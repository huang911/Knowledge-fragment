main.js

```js
calculate() {
    const screenW = getApp().globalData.systemInfo.screenWidth;
    const FONT_SIZE = 14;
    const MOVE_TIMES = 10; // 每屏移动时间
    const { scrollList } = this.data;
    const contentLen = scrollList.reduce((acc,item) => (acc += item, acc), []).length;
    const intervalItem = 30; // 每条数据的间隔
    const contentW = contentLen * FONT_SIZE + (scrollList.length - 1) * intervalItem; // 获取文本宽度；
    let allT = (contentW / screenW) * MOVE_TIMES; // 计算文字有几屏
    allT = allT < 10 ? 10: allT;
    this.setData({
      marqueeW: contentW,
      allT
    })
  },
```
