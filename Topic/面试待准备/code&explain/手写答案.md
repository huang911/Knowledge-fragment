```js

var a = {n: 1};
var b = a;
a.x = a = {n: 2};
console.log(a.x)    //undefined
console.log(b.x)    //{n: 2}

var a = {n: 1};
var b = a;
a = a.x = {n: 2};

console.log(a.x)    //undefined
console.log(b.x)   //{n: 2}

b.x.n = 4;
console.log(a.n);  4
```
